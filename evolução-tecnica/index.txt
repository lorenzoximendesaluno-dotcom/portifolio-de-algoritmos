    <script>
        // Inicializa ícones elegantes
        lucide.createIcons();

        // Sistema de abas moderno
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(t => {
                t.classList.add('hidden-tab');
                t.classList.remove('tab-transition');
            });
            
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('active');
                btn.classList.add('border-transparent', 'text-isa-textMuted');
                btn.classList.remove('text-isa-textMain');
            });

            const activeTab = document.getElementById(`tab-${tabId}`);
            activeTab.classList.remove('hidden-tab');
            // Força reflow para re-engatilhar animação CSS
            void activeTab.offsetWidth;
            activeTab.classList.add('tab-transition');
            
            const btn = document.getElementById(`btn-${tabId}`);
            btn.classList.add('active', 'text-isa-textMain');
            btn.classList.remove('border-transparent', 'text-isa-textMuted');
        }

        // Relógio
        function updateTime() {
            const now = new Date();
            document.getElementById('sys-clock').textContent = now.toLocaleTimeString('pt-BR');
            document.getElementById('sys-date').textContent = now.toLocaleDateString('pt-BR');
        }
        setInterval(updateTime, 1000);
        updateTime();

        // Gráfico Premium (Chart.js com efeitos SaaS)
        const ctx = document.getElementById('trend-chart').getContext('2d');
        
        // Criar gradiente elegante sob a linha
        const gradient = ctx.createLinearGradient(0, 0, 0, 400);
        gradient.addColorStop(0, 'rgba(0, 209, 143, 0.2)');
        gradient.addColorStop(1, 'rgba(0, 209, 143, 0)');

        let chartData = Array(30).fill(25.32);
        
        const trendChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: Array(30).fill(''),
                datasets: [{
                    label: 'MW',
                    data: chartData,
                    borderColor: '#00D18F', // Verde Moderno
                    borderWidth: 2,
                    backgroundColor: gradient,
                    fill: true,
                    pointRadius: 0,
                    pointHoverRadius: 6,
                    pointHoverBackgroundColor: '#fff',
                    pointHoverBorderColor: '#00D18F',
                    pointHoverBorderWidth: 2,
                    tension: 0.4 // Linha suave e elegante (SaaS style)
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                animation: false,
                interaction: {
                    intersect: false,
                    mode: 'index',
                },
                plugins: { 
                    legend: { display: false },
                    tooltip: {
                        backgroundColor: '#131A22',
                        titleColor: '#8B949E',
                        bodyColor: '#E6EDF3',
                        borderColor: 'rgba(255,255,255,0.1)',
                        borderWidth: 1,
                        padding: 10,
                        displayColors: false,
                        callbacks: {
                            label: function(context) { return context.parsed.y.toFixed(2) + ' MW'; }
                        }
                    }
                },
                scales: {
                    x: { display: false },
                    y: { 
                        min: 24, max: 27,
                        grid: { color: 'rgba(255,255,255,0.03)', drawBorder: false },
                        ticks: { color: '#8B949E', font: { family: 'Inter', size: 10 }, padding: 10 }
                    }
                }
            }
        });

        // Simulação de Dados
        let metrics = { mw: 25.32, pt: 65.4, hz: 60.02 };
        setInterval(() => {
            metrics.mw += (Math.random() - 0.45) * 0.1;
            metrics.pt += (Math.random() - 0.5) * 0.2;
            metrics.hz += (Math.random() - 0.5) * 0.02;

            if(metrics.mw > 26.5) metrics.mw = 26.0;

            document.getElementById('dash-mw').textContent = metrics.mw.toFixed(2);
            document.getElementById('dash-pt').textContent = metrics.pt.toFixed(1);
            document.getElementById('dash-hz').textContent = metrics.hz.toFixed(2);
            
            chartData.shift();
            chartData.push(metrics.mw);
            trendChart.update('none');
        }, 1000);
    </script>
